Quick Start Guide

Basic Custodial Wallet

import { 
  Custodial_Wallet, 
  BIP44_CONSTANTS, 
  DERIVATION_PATHS,
  BECH32 
} from 'j-bitcoin';

// Generate new wallet with built-in constants
const [mnemonic, wallet] = Custodial_Wallet.fromRandom('main', 'secure-pass');
console.log('Mnemonic:', mnemonic);
console.log('Master Address:', wallet.address);

// Use convenience methods for standard Bitcoin addresses
wallet.deriveReceivingAddress(0)    // m/44'/0'/0'/0/0
      .deriveReceivingAddress(1)    // m/44'/0'/0'/0/1
      .deriveChangeAddress(0)       // m/44'/0'/0'/1/0
      .deriveTestnetAddress(0);     // m/44'/1'/0'/0/0

// Get comprehensive wallet summary
const summary = wallet.getSummary();
console.log(`Network: ${summary.network}`);
console.log(`Receiving: ${summary.receivingAddresses}, Change: ${summary.changeAddresses}`);

// Convert legacy address to SegWit
const legacyAddr = wallet.address;
const segwitAddr = BECH32.to_P2WPKH(legacyAddr);
console.log('SegWit Address:', segwitAddr);

// Sign and verify messages
const [signature, recoveryId] = wallet.sign("Hello Bitcoin!");
console.log('Signature valid:', wallet.verify(signature, "Hello Bitcoin!"));

Complete Threshold Signatures (2-of-3)

import { Non_Custodial_Wallet, ThresholdSignature } from 'j-bitcoin';

// Create 2-of-3 threshold wallet
const wallet = Non_Custodial_Wallet.fromRandom("main", 3, 2);

// Get shares for secure distribution
const shares = wallet._shares;
console.log('Distribute these shares securely:', shares);

// Generate threshold signature
const signature = wallet.sign("Multi-party authorization required");
console.log('Threshold signature:', signature.serialized_sig);
console.log('Recovery ID:', signature.recovery_id);

// Verify threshold signature
const isValid = wallet.verify(signature.sig, signature.msgHash);
console.log('Signature valid:', isValid);

// Get detailed threshold information

// Reconstruct wallet from shares (any 2 of the 3)

  [shares[0], shares[2]], 2);
console.log('Reconstructed address matches:', 
  reconstructed.address === wallet.address);

import {

  derive,

  [3, polynomial.evaluate(3)],

// Custom derivation paths

const [enterpriseKeys, enterpriseFormat] = derive(enterprisePath, masterKeys.HDpri, format);

// Low-level cryptographic operations
const message = Buffer.from("Advanced cryptographic message", 'utf8');
const hash = rmd160(message);
const encoded = b58encode(hash);

// Key manipulation
const wifPrivateKey = "L1vHfV6GUbMJSvFaqjnButzwq5x4ThdFaotpUgsfScwMNKjdGVuS";
const rawPrivateKey = privateKey_decode(wifPrivateKey);

// Advanced threshold signature operations
class CorporateVault extends ThresholdSignature {
  constructor(boardMembers, requiredApprovals) {
    super(boardMembers, requiredApprovals);
    this.auditLog = [];
  }
  
  authorizeTransaction(transaction) {
    const signature = this.sign(JSON.stringify(transaction));
    this.auditLog.push({
      transaction,
      signature: signature.serialized_sig,
      timestamp: Date.now(),
      participants: this.threshold
    });
    return signature;
  }
  
  getAuditTrail() {
    return this.auditLog;
  }
}

  amount: 1000000,
  recipient: "1BvBMSEYstWetqTFn5Au4m4GFg7xJaNVN2",
  purpose: "Quarterly dividend payment"
};

const authorization = corporateVault.authorizeTransaction(transaction);
console.log('Corporate authorization:', authorization.serialized_sig);

fromSeed BIP32

Generates BIP32 master keys from a cryptographic seed using HMAC-SHA512 with "Bitcoin seed" as the key.

Function Signature

fromSeed(seed, net?)

Creates master private and public keys from a hex-encoded seed according to BIP32 specification.

Parameters
  • seed string - Hex-encoded cryptographic seed (typically 128-512 bits from BIP39)
  • net string optional - Network type ('main' or 'test') [default: 'main']
Returns

[HDKeys, SerializationFormat] - HD key pair and internal serialization format

import { fromSeed } from 'j-bitcoin';

// Standard BIP39 seed (512 bits / 128 hex characters)
const seed = "000102030405060708090a0b0c0d0e0f101112131415161718191a1b1c1d1e1f";
const [hdKeys, format] = fromSeed(seed, "main");

console.log('Master Private Key (xprv):', hdKeys.HDpri);
console.log('Master Public Key (xpub):', hdKeys.HDpub);

// Access internal components
console.log('Private Key:', format.privKey.key.toString('hex'));
console.log('Public Key:', format.pubKey.key.toString('hex'));
console.log('Chain Code:', format.chainCode.toString('hex'));

// Testnet master keys
const [testKeys, testFormat] = fromSeed(seed, "test");
console.log('Testnet Master Private (tprv):', testKeys.HDpri);
console.log('Testnet Master Public (tpub):', testKeys.HDpub);
Throws
  • Error - If seed results in invalid private key (extremely rare: ~1 in 2^127)
  • Error - If seed is not valid hexadecimal
  • Error - If network parameter is not recognized

Algorithm Details

BIP32 Master Key Generation Process
  1. HMAC-SHA512 Computation: Uses "Bitcoin seed" as HMAC key and input seed as data
  2. Key Material Split: Divides 512-bit result into 256-bit private key (IL) and 256-bit chain code (IR)
  3. Validation: Ensures IL is valid (non-zero and less than curve order)
  4. Public Key Derivation: Computes corresponding compressed public key using secp256k1
  5. Serialization: Creates extended key format with network-specific version bytes

derive BIP32

Derives child keys from parent keys using BIP32 hierarchical deterministic algorithm with support for both hardened and non-hardened derivation.

Function Signature

derive(path, key?, serialization_format)

Performs BIP32 child key derivation from a parent key using derivation paths.

Parameters
  • path string - BIP32 derivation path (e.g., "m/44'/0'/0'/0/0")
  • key string optional - Parent extended key in xprv/xpub or tprv/tpub format [default: '']
  • serialization_format Object - Parent key's serialization metadata
Returns

[DerivedKeys, ChildSerializationFormat] - Tuple of derived keys and child serialization format

import { fromSeed, derive } from 'j-bitcoin';

// Start with master keys
const [masterKeys, masterFormat] = fromSeed("000102030405060708090a0b0c0d0e0f", "main");

// Standard BIP44 Bitcoin account derivation
const [accountKeys, accountFormat] = derive("m/44'/0'/0'", masterKeys.HDpri, masterFormat);

// Derive receiving addresses
const [receiving0, receiving0Format] = derive("m/0/0", accountKeys.HDpri, accountFormat);
const [receiving1, receiving1Format] = derive("m/0/1", accountKeys.HDpri, accountFormat);

// Derive change addresses
const [change0, change0Format] = derive("m/1/0", accountKeys.HDpri, accountFormat);

console.log('Account xprv:', accountKeys.HDpri);
console.log('First receiving address derived from:', receiving0.HDpub);
console.log('Derivation depth:', receiving0Format.depth); // 5
Throws
  • Error - "Public Key can't derive from hardened path" if attempting hardened derivation from public key
  • Error - If path format is invalid or contains non-numeric components
  • Error - If derived key is invalid (extremely rare: ~1 in 2^127)

Derivation Types

Hardened vs Non-Hardened Derivation
Hardened Derivation (index 2, marked with ')
  • Requires private key access
  • Provides security isolation between parent and child
  • Used for account-level and above in BIP44
  • Example: "m/44'/0'/0'" - all hardened
Non-Hardened Derivation (index < 2)
  • Can derive from public key only
  • Enables watch-only wallets
  • Used for address-level derivation
  • Example: "m/0/1/2" - all non-hardened
// Hardened derivation (requires private key)
const [hardened, _] = derive("m/44'/0'/0'", masterKeys.HDpri, masterFormat);

// Non-hardened derivation (can use public key)
const [nonHardened, __] = derive("m/0/1/2", accountKeys.HDpub, accountFormat);

// Mixed derivation
const [mixed, ___] = derive("m/44'/0'/0'/0/5", masterKeys.HDpri, masterFormat);

// This will throw an error - can't derive hardened from public key
try {
  const [error, ____] = derive("m/0'", masterKeys.HDpub, masterFormat);
} catch (e) {
  console.log(e.message); // "Public Key can't derive from hardened path"
}

Common Derivation Patterns

Standard Bitcoin Derivation Paths 
// BIP44 Bitcoin Legacy (P2PKH)
const [bip44Account, bip44Format] = derive("m/44'/0'/0'", masterKeys.HDpri, masterFormat);
const [bip44Receiving, _] = derive("m/0/0", bip44Account.HDpri, bip44Format);
const [bip44Change, __] = derive("m/1/0", bip44Account.HDpri, bip44Format);

// BIP49 SegWit-nested-in-P2SH
const [bip49Account, bip49Format] = derive("m/49'/0'/0'", masterKeys.HDpri, masterFormat);
const [bip49Address, ___] = derive("m/0/0", bip49Account.HDpri, bip49Format);

function deriveComplexPath(masterKeys, masterFormat, fullPath) {

Algorithm Details

BIP39 Mnemonic Generation Process
  1. Entropy Generation: 16 bytes (128 bits) of cryptographically secure random data
  2. Checksum Calculation: SHA256 hash of entropy, take first 4 bits
  3. Binary Concatenation: Entropy (128 bits) + checksum (4 bits) = 132 bits total
  4. Word Mapping: Split 132 bits into 12 groups of 11 bits each
  5. Word Selection: Each 11-bit value maps to word from BIP39 English wordlist
// Understanding the generation process
import { randomBytes, createHash } from 'crypto';

function demonstrateMnemonicGeneration() {
  // Step 1: Generate 16 bytes of entropy
  const entropy = randomBytes(16);
  console.log('Entropy (hex):', entropy.toString('hex'));
  
  // Step 2: Calculate checksum
  const hash = createHash('sha256').update(entropy).digest();
  const checksumBits = hash[0].toString(2).padStart(8, '0').slice(0, 4);
  console.log('Checksum bits:', checksumBits);
  
  // Step 3: Create binary string
  const entropyBinary = entropy.reduce((str, byte) => 
    str + byte.toString(2).padStart(8, '0'), '');
  const fullBinary = entropyBinary + checksumBits;
  console.log('Full binary length:', fullBinary.length); // 132 bits
  
  // Step 4: Split into 11-bit groups and map to words
  const wordIndices = [];
  for (let i = 0; i < 12; i++) {
    const elevenBits = fullBinary.slice(i * 11, (i + 1) * 11);
    const wordIndex = parseInt(elevenBits, 2);
    wordIndices.push(wordIndex);
  }
  
  console.log('Word indices:', wordIndices);
  // Each index should be 0-2047 (2^11 = 2048 possible words)
}

Security Properties

Cryptographic Security Features
  • Entropy Source: Uses Node.js crypto.randomBytes() for cryptographically secure randomness
  • Entropy Amount: 128 bits provides ~2^128 possible mnemonics
  • Checksum Protection: 4-bit checksum provides error detection
  • Wordlist Design: BIP39 English wordlist avoids ambiguous words
  • Deterministic: Same entropy always produces same mnemonic

// Example 4: Mnemonic entropy analysis
function analyzeMnemonicEntropy(mnemonic) {
  const words = mnemonic.split(' ');
  
  if (words.length !== 12) {
    return { valid: false, error: 'Not 12 words' };
  }
  
  // Calculate effective entropy
  const possibleCombinations = Math.pow(2048, 12); // 2048^12
  const effectiveEntropy = Math.log2(possibleCombinations);
  
  return {
    valid: true,
    wordCount: words.length,
    possibleCombinations: possibleCombinations.toExponential(2),
    effectiveEntropy: effectiveEntropy.toFixed(1) + ' bits',
    checksumValid: bip39.checkSum(mnemonic)
  };
}
